    public qid: number,
    /**
     * type of quiz (e.g.: 'MC' for multiple-choice)
     */
    public qtype: string,
    /**
     * text/question of quiz
     */
    public qtext: string,
    /**
     * tags-array to define which anser is true or false

     * difficulty of quiz

      aid: number,
      text: string,

    }],
  ) {}
